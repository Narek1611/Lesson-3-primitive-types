>### **3. swap1**

```

let a = 10;

let c;
c = a; //c=10

b = c;

console.log(a);
console.log(b);
```

>### **4. swap2**

```
"use strict";


let a = 10;
b = 3;

b = a - b; // b=10 a=13
a = a - b; // a=3 b=10
console.log(a);
console.log(b);
```
